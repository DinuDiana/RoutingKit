9 inline double geo_dist(double lat_a, double lon_a, double lat_b, double lon_b){
10 
11 // The formula used in this function was derived as following:
12 //
13 // lat_a = input
14 // lon_a = input
15 // lat_b = input
16 // lon_b = input
17 //
18 // ----- Step 1: Convert from degree to radian
19 //
20 // lat_a /= 180
21 // lat_a *= pi
22 // lon_a /= 180
23 // lon_a *= pi
24 // lat_b /= 180
25 // lat_b *= pi
26 // lon_b /= 180
27 // lon_b *= pi
28 //
29 // ----- Step 2: Convert to 3D unit vectors
30 //
31 // x_a = cos(lon_a) * cos(lat_a)
32 // y_a = sin(lon_a) * cos(lat_a)
33 // z_a = sin(lat_a)
34 //
35 // x_b = cos(lon_b) * cos(lat_b)
36 // y_b = sin(lon_b) * cos(lat_b)
37 // z_b = sin(lat_b)
38 //
39 // ----- Step 3: Compute scalar product
40 //
41 // c = x_a * x_b + y_a * y_b + z_a * z_b
42 //
43 // ----- Step 4: acos(c) is length in radian, scaling up by earth radius gives distance
44 //
45 // length = earth_radius*acos(c)
46 // Output length
47 //
48 // ----- Steps 2 to 4 merged in one formula is
49 //
50 // length = R*acos(
51 // cos(lon_a) * cos(lat_a) * cos(lon_b) * cos(lat_b) +
52 // sin(lon_a) * cos(lat_a) * sin(lon_b) * cos(lat_b) +
53 // sin(lat_a) * sin(lat_b)
54 // )
55 //
56 // ----- It can be simplified as follows using standard trigonometric identities and basic math operations
57 // See https://en.wikipedia.org/wiki/List_of_trigonometric_identities
58 //
59 // = R*acos(
60 // cos(lat_a) * cos(lat_b) * (
61 // cos(lon_a) * cos(lon_b) +
62 // sin(lon_a) * sin(lon_b)
63 // )+
64 // sin(lat_a) * sin(lat_b)
65 // ) = R*acos(
66 // cos(lat_a) * cos(lat_b) * (
67 // 1/2 * (cos(lon_a - lon_b) + cos(lon_a + lon_b)) +
68 // 1/2 * (cos(lon_a - lon_b) - cos(lon_a + lon_b))
69 // )+
70 // sin(lat_a) * sin(lat_b)
71 // ) = R*acos(
72 // 1/2 * cos(lat_a) * cos(lat_b) * (
73 // cos(lon_a - lon_b) + cos(lon_a + lon_b) +
74 // cos(lon_a - lon_b) - cos(lon_a + lon_b)
75 // )+
76 // sin(lat_a) * sin(lat_b)
77 // ) = R*acos(
78 // 1/2 * cos(lat_a) * cos(lat_b) * (
79 // cos(lon_a - lon_b) +
80 // cos(lon_a - lon_b)
81 // )+
82 // sin(lat_a) * sin(lat_b)
83 // ) = R*acos(
84 // cos(lat_a) * cos(lat_b) * cos(lon_a - lon_b) +
85 // sin(lat_a) * sin(lat_b)
86 // ) = R*acos(
87 // 1/2 * (cos(lat_a - lat_b) + cos(lat_a + lat_b)) * cos(lon_a - lon_b) +
88 // 1/2 * (cos(lat_a - lat_b) - cos(lat_a + lat_b))
89 // ) = R*acos(
90 // 1/2 * (
91 // (cos(lat_a - lat_b) + cos(lat_a + lat_b)) * cos(lon_a - lon_b) +
92 // (cos(lat_a - lat_b) - cos(lat_a + lat_b))
93 // )
94 // )
95 
96  const double pi_div_180 = 3.14159265359/180.0;
97  const double earth_radius = 6371000.785; // in meter
98 
99  // To help the auto-vectorizer figure out that using SIMD is a good idea here, we use a local array with a constant size instead of variables.
100  // GCC 5's auto vectorizer requires that the array has a power of two size.
101  // We thus add a fourth dummy value.
102  //
103  // If the code was vectorized then the compiler would have to insert this dummy value implicitly. We are helping it by doing it explicitly.
104  // If SIMD is disabled, the array should be converted by the optimizer to 4 variables. The dummy variable should subsequently be eliminated using dead code elimination.
105  double vec[4] = {
106  lat_a - lat_b,
107  lat_a + lat_b,
108  lon_a - lon_b,
109  0
110  };
111 
112  // It is important that this loop has four iterations, as otherwise the auto-vectorizer does not do its job.
113  for(unsigned i=0; i<4; ++i)
114  vec[i] = cos(vec[i]*pi_div_180);
115 
116  double len = (vec[0] + vec[1]) * vec[2] + vec[0] - vec[1];
117  len *= 0.5;
118  len = acos(len);
119  len *= earth_radius;
120 
121  // len is in meter
122 
123  return len;
124 }
